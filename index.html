        function renderQuiz() {

            const root = document.getElementById('quiz-root');

            quizData.forEach((data, index) => {

                const card = document.createElement('div');

                card.className = 'question-card';

                card.innerHTML = `

                    <div class="question-text">${data.q}</div>

                    <div class="options-container" id="opts-${data.id}"></div>

                    <div class="explanation-panel" id="exp-panel-${data.id}">

                        <h4>Option Explanations:</h4>

                        <div id="exp-content-${data.id}"></div>

                    </div>

                `;

                root.appendChild(card);


                const optContainer = document.getElementById(`opts-${data.id}`);

                const expContent = document.getElementById(`exp-content-${data.id}`);

                const expPanel = document.getElementById(`exp-panel-${data.id}`);


                Object.entries(data.options).forEach(([key, text]) => {

                    const btn = document.createElement('button');

                    btn.className = 'option-btn';

                    btn.innerHTML = `<span class="option-letter">${key}.</span> <span>${text}</span>`;

                    

                    btn.onclick = () => {

                        handleAnswer(data.id, key, data.correct);

                    };

                    optContainer.appendChild(btn);


                    // Pre-fill explanation content

                    const expItem = document.createElement('div');

                    expItem.className = 'exp-item';

                    expItem.innerHTML = `<span class="exp-key">Option ${key}:</span> ${data.explanations[key]}`;

                    expContent.appendChild(expItem);

                });

            });

        }


        function handleAnswer(qId, selectedKey, correctKey) {

            const container = document.getElementById(`opts-${qId}`);

            const buttons = container.getElementsByClassName('option-btn');

            const panel = document.getElementById(`exp-panel-${qId}`);


            Array.from(buttons).forEach(btn => {

                btn.classList.add('disabled');

                const btnKey = btn.querySelector('.option-letter').innerText.replace('.', '');

                

                if (btnKey === correctKey) {

                    btn.classList.add('correct');

                } else if (btnKey === selectedKey) {

                    btn.classList.add('wrong');

                }

            });


            panel.style.display = 'block';

        }


        renderQuiz();

    </script>
